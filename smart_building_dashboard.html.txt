<script>

function predictTemp(){

let outside = document.getElementById("tempInput").value;

if(outside == ""){
alert("Please enter temperature");
return;
}

let inside = 25 - (outside/5);

document.getElementById("tempResult").innerHTML =
"Recommended Indoor Temperature: " + inside.toFixed(1) + " °C";

}

function energyCheck(){

let devices = ["AC","Washing Machine","Heater"];

document.getElementById("energyResult").innerHTML =
"High Energy Consumption Devices: " + devices.join(", ") +
". Suggestion: Reduce usage time when not required.";

}

function checkWall(){

document.getElementById("wallResult").innerHTML =
"Possible cracks or leakage detected in the wall. Maintenance recommended.";

}
